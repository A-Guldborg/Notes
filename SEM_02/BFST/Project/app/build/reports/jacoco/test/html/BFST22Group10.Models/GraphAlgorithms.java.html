package BFST22Group10.Models;

import java.util.*;

public class GraphAlgorithms {
    
    public static ArrayList<Vertex> shortestPath(Vertex source, Vertex destination, String modeOfTransport){
        source.setDistTo(0.0f);
        HashSet<Vertex> visited = new HashSet<>();
        HashSet<Edge> visitedEdges = new HashSet<>();
        HashMap<Vertex, Vertex> visitedFrom = new HashMap<>();
        
        PriorityQueue<Vertex> priorityQueue = new PriorityQueue<>();
        priorityQueue.add(source);
        pqIterator:
        while(!priorityQueue.isEmpty()){
            Vertex vertexFrom = priorityQueue.poll();
            for (Edge edge : vertexFrom.getNeighbors()) {
                Vertex vertexTo = edge.getTo();
                if(!edge.isTransportModeAllowed(modeOfTransport)) continue;
                if(!visited.contains(vertexTo) || vertexTo.getDistTo() > vertexFrom.getDistTo() + edge.getDistance(modeOfTransport)){
                    vertexTo.setDistTo(vertexFrom.getDistTo() + edge.getDistance(modeOfTransport)); // relax edge
                    visitedFrom.put(vertexTo, vertexFrom);
                    if(!priorityQueue.contains(vertexTo)){
                        priorityQueue.add(vertexTo);
                    }
                }
                if(vertexTo == destination) {
                    for (Vertex vertex : visitedFrom.values()) {
                        vertex.setDistTo(0f);
                    }
                    destination.setDistTo(0f);
                    break pqIterator;
                }
                visitedEdges.add(edge);
            }
            visited.add(vertexFrom);
        }
        ArrayList<Vertex> path = new ArrayList<>();
        Vertex current = destination;
        while(current != null){
            path.add(current);
            current = visitedFrom.get(current);
        }
        ArrayList<Vertex> sourceToDestination = new ArrayList<>();
        for (int i = path.size() - 1; i >= 0; i--) {
            sourceToDestination.add(path.get(i));
        }
        return sourceToDestination;
    }

        pqIterator:
        while(!priorityQueue.isEmpty()){
            Vertex vertexFrom = priorityQueue.poll();
            for (Edge edge : vertexFrom.getNeighbors()) {
                Vertex vertexTo = edge.getTo();

                if(!visited.contains(vertexTo) || vertexTo.getDistTo() > vertexFrom.getDistTo() + edge.getDistance(modeOfTransport)){

            }

        }
        
        return visitedEdges;
    }
