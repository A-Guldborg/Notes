package BFST22Group10;

import javafx.scene.canvas.GraphicsContext;

import java.util.ArrayList;
import java.util.HashMap;

public class Way extends Member {
    private ArrayList<Node> nodesOfWay = new ArrayList<>();
    
    public Way(long id, HashMap<String, String> tags, ArrayList<Node> nodesOfWay) {
        super(id, tags);
        this.nodesOfWay = nodesOfWay;
    }
    
    public void draw(GraphicsContext canvas){
        canvas.setLineWidth(1/Math.sqrt(canvas.getTransform().determinant()));
        for (int i = 0; i < nodesOfWay.size() - 1; i++) {
            canvas.beginPath();
            canvas.moveTo(nodesOfWay.get(i).getLon(), -nodesOfWay.get(i).getLat());
            canvas.lineTo(nodesOfWay.get(i + 1).getLon(), -nodesOfWay.get(i + 1).getLat());
            canvas.stroke();
        }
    }
}
