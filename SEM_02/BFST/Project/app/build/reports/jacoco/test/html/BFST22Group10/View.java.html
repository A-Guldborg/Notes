package BFST22Group10;

import javafx.fxml.FXMLLoader;
import javafx.stage.Stage;

import java.io.IOException;
import java.util.List;

public class View {
    public final MapCanvas mapCanvas;
    
    public View(Stage primaryStage) throws IOException {
        FXMLLoader loader = new FXMLLoader(getClass().getResource("/View.fxml"));
        primaryStage.setScene(loader.load());
        Controller controller = loader.getController();
        mapCanvas = controller.mapCanvas;
        primaryStage.setTitle("Map");
        primaryStage.show();
    }
    
    public void drawModel(List<Way> ways){
        mapCanvas.clear();
        for (Way way : ways) {
            way.draw(mapCanvas.getGraphicsContext2D());
        }
    }
}
