package BFST22Group10;

import javax.xml.stream.XMLInputFactory;
import javax.xml.stream.XMLStreamConstants;
import javax.xml.stream.XMLStreamException;
import javax.xml.stream.XMLStreamReader;
import java.io.*;
import java.util.ArrayList;
import java.util.HashMap;

// https://docs.fileformat.com/gis/osm/

// Nodes
// Represents geographic position stored as pairs of a latitude and a longitude.
// Used to represent map features without a size, such as mountain peaks.

// Ways
// Sorted lists of nodes, signifying a polyline, or a polygon Represent linear features
// such as roads and rivers, and zones, like parking areas jungles, and parks.

// Relations
// Sorted lists of nodes and ways represent their relation like barriers and u turns
// on roads, motorways span different existing ways and areas with holes.

// Tags
// Store metadata about the map objects. Always attached to any node, way or a relation

// https://wiki.openstreetmap.org/wiki/Elements


public class Model {
    private float minlat, minlon, maxlat, maxlon;
    private HashMap<Long, Node> id2node = new HashMap<>();
    private HashMap<Long, Way> id2way = new HashMap<>();
    private HashMap<Long, Relation> id2relation = new HashMap<>();
    private ArrayList<Way> ways = new ArrayList<>();
    private HashMap<String, String> tags = new HashMap<>();
    private ArrayList<Node> nodesOfWay = new ArrayList<>();
    private long currentId;
    private float currentLon;
    private float currentLat;
    
    private float lonMultiplier = 0.56f;
    
    public Model(InputStream osmfile) throws XMLStreamException {
        loadOSM(osmfile);
    }

    private void loadOSM(InputStream inputStream) throws XMLStreamException {
        XMLStreamReader reader = XMLInputFactory.newInstance().createXMLStreamReader(new BufferedInputStream(inputStream));

        while(reader.hasNext()){
            switch (reader.next()){
                case XMLStreamConstants.START_ELEMENT:
                    String name = reader.getLocalName();
                    switch (name){
                        case "bounds":
                            minlat = Float.parseFloat(reader.getAttributeValue(null, "minlat"));
                            minlon = lonMultiplier * Float.parseFloat(reader.getAttributeValue(null, "minlon"));
                            maxlat = Float.parseFloat(reader.getAttributeValue(null, "maxlat"));
                            maxlon = lonMultiplier * Float.parseFloat(reader.getAttributeValue(null, "maxlon"));
                            break;
                        case "node":
                            tags = new HashMap<>();
                            currentId = Long.parseLong(reader.getAttributeValue(null, "id"));
                            currentLat = Float.parseFloat(reader.getAttributeValue(null, "lat"));
                            currentLon = lonMultiplier * Float.parseFloat(reader.getAttributeValue(null, "lon"));

                        case "way":
                            nodesOfWay = new ArrayList<>();
                            tags = new HashMap<>();
                            currentId = Long.parseLong(reader.getAttributeValue(null, "id"));
                            break;
                        case "nd":
                            nodesOfWay.add(id2node.get(Long.parseLong(reader.getAttributeValue(null, "ref"))));
                            break;
                        case "relation":
                            tags = new HashMap<>();
                            currentId = Long.parseLong(reader.getAttributeValue(null, "id"));
                            break;
                        case "tag":
                            String key = reader.getAttributeValue(null, "k");
                            String value = reader.getAttributeValue(null, "v");
                            tags.put(key, value);
                            break;
                        default:
                            break;
                    }
                    break;
                case XMLStreamConstants.END_ELEMENT:
                    switch (reader.getLocalName()){
                        case "way":
                            Way way = new Way(currentId, tags, this.nodesOfWay);
                            id2way.put(currentId, way);
                            ways.add(way);
                            break;
                        case "node":
                            Node node = new Node(currentId, tags, currentLat, currentLon);
                            id2node.put(currentId, node);
                            break;
                        case "relation":
                            Relation relation = new Relation(currentId, tags);
                            id2relation.put(currentId, relation);
                            break;
                    }
                default:
                    break;
            }
        }
        reader.close();
    }
    
    public float getMinLat() {
        return minlat;
    }
    
    public float getMinLon() {
        return minlon;
    }
    
    public float getMaxLat() {
        return maxlat;
    }
    
    public float getMaxLon() {
        return maxlon;
    }
    
    public HashMap<Long, Node> getNodes() {
        return id2node;
    }
    
    public ArrayList<Way> getWays() {
        return ways;
    }
}
